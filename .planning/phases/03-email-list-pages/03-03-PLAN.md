import sys; sys.path.insert(0, 'scripts')
from build import RESOURCES
target_slugs = ['oral-pathologists-email-list', 'chiropractor-email-list', 'naturopathic-doctors-email-list', 'periodontist-email-list', 'physiatrist-email-list', 'iv-infusion-therapy-email-list', 'lasik-surgeons-email-list', 'med-spa-owners-contact-list']
all_pass = True
for slug in target_slugs:
    r = [x for x in RESOURCES if x['slug'] == slug][0]
    t = r['title']
    m = r['meta_description']
    h = r['h1']
    errors = []
    if len(t) < 50 or len(t) > 60:
        errors.append(f'title len {len(t)} (need 50-60)')
    if len(m) < 120 or len(m) > 158:
        errors.append(f'meta len {len(m)} (need 120-158)')
    if 'email list' not in h.lower() and 'contact' not in h.lower():
        errors.append('H1 missing email list/contact keyword')
    if len(r.get('faqs', [])) < 3:
        errors.append(f'only {len(r.get(\"faqs\",[]))} FAQs (need 3+)')
    if len(r.get('outbound_links', [])) < 2:
        errors.append(f'only {len(r.get(\"outbound_links\",[]))} outbound links (need 2+)')
    if errors:
        print(f'FAIL {slug}: {errors}')
        all_pass = False
    else:
        print(f'PASS {slug}: title={len(t)}c, meta={len(m)}c, faqs={len(r[\"faqs\"])}, outbound={len(r[\"outbound_links\"])}')
assert all_pass, 'Some pages failed validation'
print('All 8 pages pass SEO checks')
